15 def __map_granularity_to_yahoo(self, granularity: Granularity) -> str: 

16 """ 

17 Maps internal Granularity enum to Yahoo Finance interval format. 

18 

19 Parameters: 

20 granularity (Granularity): Granularity enum value 

21 

22 Returns: 

23 (str): Yahoo Finance compatible interval string 

24 """ 

25 # Yahoo Finance intervals: 1m, 2m, 5m, 15m, 30m, 60m, 90m, 1h, 1d, 5d, 1wk, 1mo, 3mo 

26 mapping = { 

27 Granularity.ONE_MINUTE: "1m", 

28 Granularity.FIVE_MINUTES: "5m", 

29 Granularity.FIFTEEN_MINUTES: "15m", 

30 Granularity.ONE_HOUR: "60m", 

31 Granularity.SIX_HOURS: "1h", # No exact match, using 1h 

32 Granularity.ONE_DAY: "1d" 

33 } 

34 

35 if granularity not in mapping: 

36 raise ValueError(f"Granularity {granularity} not supported by Yahoo Finance API") 

37 

38 return mapping[granularity]

103 async def get_candles_for(self, symbol: str, start_date: str, end_date: str, granularity: Granularity) -> pd.DataFrame: 

104 """ 

105 Collects data from Yahoo Finance API given starting date, ending date, granularity and symbol. 

106 Works for cryptocurrencies, stocks, and indices. 

107 

108 Parameters: 

109 symbol (str): String representing unique symbol (e.g. "BTC-USD", "AAPL", "^GSPC"). 

110 start_date (str): String representing date that collected data should start from. 

111 end_date (str): String representing date that collected data should finish at. 

112 granularity (Granularity): Enum specifying resolution of collected data. 

113 

114 Raises: 

115 ValueError: If given granularity is not supported by Yahoo Finance. 

116 

117 Returns: 

118 (pd.DataFrame): Collected data frame. 

119 """ 

120 if granularity not in Granularity: 

121 raise ValueError(f"{granularity} is not a value of Granularity enum!") 

122 

123 interval = self.__map_granularity_to_yahoo(granularity) 

124 

125 # Run the Yahoo Finance download in a separate thread to avoid blocking the event loop 

126 loop = asyncio.get_event_loop() 

127 df = await loop.run_in_executor( 

128 None, 

129 lambda: yf.download( 

130 symbol, 

131 start=start_date, 

132 end=end_date, 

133 interval=interval 

134 ) 

135 ) 

136 

137 # Rename columns to match Coinbase format 

138 df = df.rename(columns={ 

139 'Low': 'low', 

140 'High': 'high', 

141 'Open': 'open', 

142 'Close': 'close', 

143 'Volume': 'volume' 

144 }) 

145 

146 df.columns = df.columns.get_level_values(0) 

147 return df
